class Solution:
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        first = 0
        while first < len(nums):
            for i in range(first+1, len(nums)):
                if nums[first] + nums[i] == target:
                  return [first, i]
            first += 1
